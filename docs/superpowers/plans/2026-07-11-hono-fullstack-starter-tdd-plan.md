### 1c — Source/import guard test

#### `packages/hono-base/src/import-guard.test.ts`

A static scan test that reads all `.ts` files under `src/` (excluding `*.test.ts`), asserts none contain `node:` or `@hono/node-server` import specifiers.

```ts
import { readdir, readFile } from 'node:fs/promises';
import { join, relative } from 'node:path';
import { describe, expect, it } from 'vitest';

const srcDir = join(import.meta.dirname);

async function walk(dir: string): Promise<string[]> {
  const entries = await readdir(dir, { withFileTypes: true });
  const files: string[] = [];
  for (const entry of entries) {
    const path = join(dir, entry.name);
    if (entry.isDirectory()) {
      files.push(...(await walk(path)));
    } else if (entry.name.endsWith('.ts') && !entry.name.endsWith('.test.ts')) {
      files.push(path);
    }
  }
  return files;
}

describe('hono-base import boundary', () => {
  it('source graph contains no node:* or @hono/node-server imports', async () => {
    const files = await walk(srcDir);
    const violations: string[] = [];
    for (const file of files) {
      const content = await readFile(file, 'utf8');
      if (/from\s+['"]node:/.test(content)) {
        violations.push(`${relative(srcDir, file)}: node:* import`);
      }
      if (/from\s+['"]@hono\/node-server['"]/.test(content)) {
        violations.push(`${relative(srcDir, file)}: @hono/node-server import`);
      }
    }
    expect(violations).toEqual([]);
  });
});
```

### 1d — Packed isolated-consumer smoke test

#### `packages/hono-base/src/packed-consumer.test.ts`

This test runs only when a packed tarball is available (via `PACKED_TARBALL` env or by packing first). It:
- Extracts the tarball to a temp dir.
- Imports `@cheshirecode/hono-base` from the extracted package.
- Calls `createBaseApp({ version: '0.0.0' }).request('/healthz')` and asserts 200 + body.
- Calls `createBaseApp({ version: '0.0.0' }).request('/version')` and asserts 200 + body.
- Lists tarball contents and asserts no `node:` or `@hono/node-server` in the dependency graph (checks `package.json` dependencies of the tarball).

```ts
import { mkdtemp, readFile, readdir, writeFile } from 'node:fs/promises';
import { createRequire } from 'node:module';
import { join, tmpdir } from 'node:path';
import { execFile } from 'node:child_process';
import { promisify } from 'node:util';
import { describe, expect, it } from 'vitest';

const execFileAsync = promisify(execFile);
const tarball = process.env.PACKED_TARBALL;

const itPacked = tarball ? it : it.skip;

describe('hono-base packed consumer', () => {
  itPacked('imports and invokes routes from a clean tarball install', async () => {
    const consumerDir = await mkdtemp(join(tmpdir(), 'hono-base-packed-'));
    await writeFile(join(consumerDir, 'package.json'), JSON.stringify({
      name: 'hono-base-packed-consumer',
      private: true,
      type: 'module',
      dependencies: { '@cheshirecode/hono-base': `file:${tarball}` }
    }));
    await execFileAsync('pnpm', ['install', '--no-frozen-lockfile'], { cwd: consumerDir });

    const { createBaseApp } = await import(
      join(consumerDir, 'node_modules', '@cheshirecode', 'hono-base', 'dist', 'index.js')
    );

    const app = createBaseApp({ version: '0.0.0' });
    const healthz = await app.request('/healthz');
    expect(healthz.status).toBe(200);
    expect(await healthz.json()).toEqual({ status: 'ok' });

    const versionRes = await app.request('/version');
    expect(versionRes.status).toBe(200);
    expect(await versionRes.json()).toEqual({ name: 'hono-base', version: '0.0.0' });
  });

  itPacked('packed dependency graph has no node:* or adapter dependency', async () => {
    const pkgJson = JSON.parse(
      await readFile(join(tarball!.replace(/\.tgz$/, ''), '..', 'package', 'package.json'), 'utf8')
        .catch(() => readFile(tarball!.replace(/\.tgz$/, '') + '.package.json', 'utf8'))
    );
    // Alternative: extract tarball and read package/package.json
    const { stdout } = await execFileAsync('tar', ['-xOf', tarball!, 'package/package.json']);
    const pkg = JSON.parse(stdout);
    const deps = Object.keys(pkg.dependencies ?? {});
    expect(deps).not.toContain('node:*');
    expect(deps).not.toContain('@hono/node-server');
  });
});
```

> Note: The packed-consumer test is skipped unless `PACKED_TARBALL` is set. The dogfood script (Phase 4) sets this env var. For local focused runs, `PACKED_TARBALL=.artifacts/release/cheshirecode-hono-base-0.0.0.tgz pnpm moon run hono-base:test` exercises it.

### 2c — Safe bootstrap serialization

#### `packages/app-react/src/shared/bootstrap.ts`

```ts
export function serializeBootstrap(payload: unknown): string {
  const json = JSON.stringify(payload);
  return json.replace(/</g, '\\u003c');
}

export function parseBootstrap(element: Element): unknown {
  const text = element.textContent ?? '';
  return JSON.parse(text);
}
```

#### `packages/app-react/src/shared/bootstrap.test.ts`

describe('bootstrap serialization', () => {
  it('escapes < so </script> cannot close the script element', () => {
    const serialized = serializeBootstrap({ html: '</script>' });
    expect(serialized).not.toContain('</script>');
    expect(serialized).toContain('\\u003c/script\\u003e');

```tsx
import type { ErrorInfo, PropsWithChildren, ReactNode } from 'react';
import { Component } from 'react';

const GENERIC_ERROR_MESSAGE = 'Something went wrong.';

export default class ErrorBoundary extends Component<
  PropsWithChildren<unknown>,
  { hasError: boolean }
> {
  constructor(props: PropsWithChildren<unknown>) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(): { hasError: boolean } {
    return { hasError: true };
  }

  componentDidCatch(_error: Error, _errorInfo: ErrorInfo): void {
    // Server-side logs may record the original error; tests must not depend on log text.
  }

  render(): ReactNode {
    if (this.state.hasError) {
      return <em className="color-error">{GENERIC_ERROR_MESSAGE}</em>;
    }
    return this.props.children;
  }
}
```

import { join, relative } from 'node:path';

const packageSrc = join(import.meta.dirname, '..');

async function walk(dir: string): Promise<string[]> {
  const entries = await readdir(dir, { withFileTypes: true });
  const files: string[] = [];
  for (const entry of entries) {
    const path = join(dir, entry.name);
    if (entry.isDirectory()) {
      files.push(...(await walk(path)));
    } else if (entry.name.endsWith('.ts') || entry.name.endsWith('.tsx')) {
      if (!entry.name.endsWith('.test.ts') && !entry.name.endsWith('.test.tsx')) {
        files.push(path);
      }
    }
  }
  return files;
}

  './server/node',

## Rejected Items (not included)

Per the council review, the following are explicitly excluded from this plan:

- **Route-precedence expansion:** No tests for route ordering beyond the basic 404/500 behavior. The spec's `notFound` + `onError` handlers are sufficient.
- **Recursive manifest traversal tests:** No tests that recursively walk the generated repo's manifest tree. The generator test asserts file inventory and dependency relationships, not deep traversal.
- **Expanded import-safe main child-process lifecycle:** No tests for the Node entry's child-process lifecycle beyond `SIGINT`/`SIGTERM` handlers. The `server/node.ts` file is not unit-tested directly; the import guard test asserts it is not transitively imported by browser entries.

## Open Questions

1. **Hono version:** The plan uses `hono@^4.9.0`. Verify this is the latest stable at implementation time. If a newer 4.x is available, use it and add to `expectedVersions` in `package-drift.mjs`.
2. **`@hono/node-server` version:** The plan uses `^1.14.0`. Same verification.
3. **Vite SSR build config:** The `vite.config.server.ts` approach is one option. An alternative is a single `vite.config.ts` with `build.lib` mode and conditional `ssr` flag. The two-config approach is simpler and more explicit. If the Vite plugin ecosystem causes issues with two configs, fall back to a single config with `--ssr` flag.
4. **`app-react` Storybook:** The existing `app-react` has Storybook config. The SSR evolution should not break Storybook. Verify `pnpm moon run app-react:build-storybook` still works after the source tree restructure. If it breaks, scope a follow-up.
5. **UnoCSS in SSR:** The `AppTree` uses UnoCSS classes. Verify `renderToString` does not crash on UnoCSS class names (it should not — they are just strings). If the SSR build needs UnoCSS plugin, add it to `vite.config.server.ts`.
